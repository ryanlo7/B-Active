  5 /**
  6     * Parses a JSON Web Token.
  7     * Source: CS144 with Professor Cho, Project 4 JWT code
  8     * @param {Object} token A JSON Web Token.
  9     * @return {String} The parsed JSON Web Token, as a string.
 10 */
 11 function parseJWT(token)
 12 {
 13     let base64Url = token.split('.')[1];
 14     let base64 = base64Url.replace(/-/g, '+').replace(/_/g, '/');
 15     return JSON.parse(atob(base64));
 16 }

 55     /**

 64 			this.userService.fetchUser()
 65 				.subscribe(user => {

 67 				});
 68 		}
 69 	}

 73         * @return {Void}
 74     */

 76 		this.userService.changeName(this.user.userId, this.user.name);
 77 		this.userService.updateActivities(this.user.userId, this.user.activities);
 78 		this.userService.updateAvailability(this.user.userId, this.user.availability);

118     /**
119         * Populates the availability table with half-hour times.
120         * @return {Void}
121     */
122 	fillTableHeadings(): void {
123 		let hour = 0;
124 		let minutes = "00";
125 
126 		for (let i = 0; i < 48; i ++) {
127 			this.tableHeadings.push(`${hour}:${minutes}`);
128 			hour = (i % 2 == 1) ? hour + 1 : hour;
129 			minutes = (minutes == "00") ? "30" : "00";
130 		}
131 	}
132 
133     /**
134         * Calculates the rating of the user.
135         * @return {Array} An array of numbers, of length 5, for which the appropriate number of entries are full.
136     */
137 	calculateRating() {
138 		let ratings = this.user.rating;
139 		if (ratings.numRatings === 0) {
140 			return 0;
141 		}
142 		let avg = Math.ceil(ratings.scoreSum / ratings.numRatings);
143 		return Array(avg).fill(5);
144 	}
