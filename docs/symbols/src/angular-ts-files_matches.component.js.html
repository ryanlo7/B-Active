  4 // Source: CS144 with Professor Cho, Project 4 JWT code
  5 function parseJWT(token)
  6 {
  7     let base64Url = token.split('.')[1];
  8     let base64 = base64Url.replace(/-/g, '+').replace(/_/g, '/');
  9     return JSON.parse(atob(base64));
 10 }

 17 /**
 18     * A class representing the component which allows the user to see her matches.

 20 */
 21 export class MatchesComponent implements OnInit {
 22 	user: User;
 23 	userId: number;
 24 	matchedEvents: Event[];
 25 	pendingEvents: Event[];
 26 	confirmedEvents: Event[];
 27 
 28     /**
 29         * Creates a MatchesComponent.
 30         * @param {UserService} userService The userService.
 31     */
 32 	constructor(private userService: UserService) {
 33 		this.userId = parseJWT(document.cookie).userId;
 34 	}

 36     /**
 37         * Function which runs at MatchesComponent's initialisation.
 38         * Get the user from the API if it has not been fetched.
 39         * Do the same for matched events, pending events, confirmed events.
 40         * @return {Void}
 41     */
 42 	ngOnInit() {
 43 		this.user = this.userService.getUser(this.userId);
 44 		if (this.user == null) {
 45 			this.userService.fetchUser()
 46 				.subscribe(user => {
 47 					this.user = this.userService.getUser(this.userId);
 48 				});
 49 		}
 50 
 51 		this.matchedEvents = this.userService.getMatchedEvents();
 52 		if (this.matchedEvents == null) {
 53 			this.userService.fetchMatchedEvents(this.userId)
 54 				.subscribe(matches => {
 55 					this.matchedEvents = this.userService.getMatchedEvents();
 56 					console.log(matches);
 57 				});
 58 		}
 59 
 60 		this.pendingEvents = this.userService.getPendingEvents();
 61 		if (this.pendingEvents == null) {
 62 			this.userService.fetchPendingEvents(this.userId)
 63 				.subscribe(pending => {
 64 					this.pendingEvents = this.userService.getPendingEvents();
 65 					console.log(pending);
 66 				});
 67 		}
 68 
 69 		this.confirmedEvents = this.userService.getConfirmedEvents();
 70 		if (this.confirmedEvents == null) {
 71 			this.userService.fetchConfirmedEvents(this.userId)
 72 				.subscribe(confirmed => {
 73 					this.confirmedEvents = this.userService.getConfirmedEvents();
 74 					console.log(confirmed);
 75 				});
 76 		}
 77 	}

 79 	/**
 80         * Given a pending event, return true if user has not accepted
 81         * @param {number} eventId The ID of the event.
 82         * @return {boolean} True if user has not accepted this event.
 83     */
 84 	hasNotAccepted(eventId: number): boolean {
 85 		for(let i = 0; i < this.pendingEvents.length; i++) {
 86 			if (eventId == this.pendingEvents[i].eventId) {
 87 				if (this.pendingEvents[i].acceptedIds.indexOf(this.userId) > -1) {
 88 					return false;
 89 				} else {
 90 					return true;
 91 				}
 92 			}
 93 		}
 94 
 95 		return false;
 96 	}

 98     /**

100         * @param {number} userId The ID of the user.

102     */
103 	getUserName(userId: number): String {
104         var user: User;
105         user = this.userService.getUser(userId);
106         if (user == null) {
107             return null;
108         }
109 
110         return user.name;
111     }

113     /**
114         * Gets the other participant in an event.
115         * @param {Array<number>} userId The IDs of the users in the event.
116         * @return {number} The ID of the other participant in an event.
117     */
118     otherParticipant(invitedIds: number[]): number {
119     	if (invitedIds[0] == this.userId) {
120     		return invitedIds[1];
121     	}
122 
123     	return invitedIds[0];
124     }

126     /**
127         * Adds the userId to the accepted array. If all invited have accepted,
128     	* changes the status to confirmed. Otherwise, sets the status of the event
129     	* to pending or invited.
130         * @param {number} eventId The ID of the event.
131         * @param {number} userId The ID of the user.
132         * @return {Void}
133     */
134 	acceptEvent(eventId: number, userId: number): void {
135 		// search through matched
136 		let found: boolean = false;
137 		for(let i = 0; i < this.matchedEvents.length; i++) {
138 			if (this.matchedEvents[i].eventId == eventId) {
139 				found = true;
140 				console.log(this.matchedEvents[i]);
141 				let newEvent: Event = this.matchedEvents[i];
142 				newEvent.acceptedIds.push(userId);
143 				newEvent.status = "pending";
144 				//this.pendingEvents.push(newEvent);
145 				this.userService.updateEvent(newEvent);
146 				console.log(newEvent);
147 				this.matchedEvents.splice(i, 1);
148 				break;
149 			}
150 		}
151 
152 		if (!found) {
153 			// serach through pending. if accepted, this means moved to confirmed
154 			for(let i = 0; i < this.pendingEvents.length; i++) {
155 				if (this.pendingEvents[i].eventId == eventId) {
156 					let newEvent: Event = this.pendingEvents[i];
157 					newEvent.acceptedIds.push(userId);
158 					newEvent.status = "confirmed";
159 					this.userService.updateEvent(newEvent);
160 					this.pendingEvents.splice(i, 1);
161 					break;
162 				}
163 			}
164 		}
165 
166 		console.log(eventId);
167 
168 	}
169 
170     /**
171         * Removes the userId from the accepted array. Sets the status of the event
172     	* to pending or just matched.
173         * @param {number} eventId The ID of the event.
174         * @return {Void}
175     */
176 	declineEvent(eventId: number): void {
177 		let found: boolean = false;
178 		for(let i = 0; i < this.matchedEvents.length; i++) {
179 			if (this.matchedEvents[i].eventId == eventId) {
180 				found = true;
181 				this.userService.deleteEvent(eventId);
182 				this.matchedEvents.splice(i, 1);
183 				break;
184 			}
185 		}
186 
187 		if (!found) {
188 			// serach through pending. if accepted, this means moved to confirmed
189 			for(let i = 0; i < this.pendingEvents.length; i++) {
190 				if (this.pendingEvents[i].eventId == eventId) {
191 					this.userService.deleteEvent(eventId);
192 					this.pendingEvents.splice(i, 1);
193 					break;
194 				}
195 			}
196 		}
197 	}
