  6 /**
  7     * Parses a JSON Web Token.
  8     * Source: CS144 with Professor Cho, Project 4 JWT code
  9     * @param {Object} token A JSON Web Token.
 10     * @return {String} The parsed JSON Web Token, as a string.
 11 */
 12 function parseJWT(token)
 13 {
 14     let base64Url = token.split('.')[1];
 15     let base64 = base64Url.replace(/-/g, '+').replace(/_/g, '/');
 16     return JSON.parse(atob(base64));
 17 }

 68     /**
 69         * Function which runs at EditComponent's initialisation.
 70         * Get the user from the API if it has not been fetched.
 71         * @return {Void}
 72     */
 73 	ngOnInit() {
 74 		this.userId = parseJWT(document.cookie).userId;
 75 
 76 		// + converts to a number
 77 		const id = +this.route.snapshot.paramMap.get('id');
 78 		if (id != null && id != this.userId) {
 79 			this.userId = id;
 80 			this.publicView = true;
 81 		}
 82 		this.fillTableHeadings();
 83 		this.user = this.userService.getUser(this.userId);
 84 		if (this.user == null) {
 85 			this.userService.fetchUser()
 86 				.subscribe(user => {
 87 					this.user = this.userService.getUser(this.userId);
 88 				});
 89 		}
 90 	}

 92     /**
 93         * Populates the availability table with half-hour times.
 94         * @return {Void}
 95     */
 96 	fillTableHeadings(): void {
 97 		let hour = 0;
 98 		let minutes = "00";
 99 
100 		for (let i = 0; i < 48; i ++) {
101 			this.tableHeadings.push(`${hour}:${minutes}`);
102 			hour = (i % 2 == 1) ? hour + 1 : hour;
103 			minutes = (minutes == "00") ? "30" : "00";
104 		}
105 	}
106 
107     /**
108         * Calculates the rating of the user.
109         * @return {Array} An array of numbers, of length 5, for which the appropriate number of entries are full.
110     */
111 	calculateRating() {
112 		let ratings = this.user.rating;
113 		if (ratings.numRatings === 0) {
114 			return 0;
115 		}
116 		let avg = Math.ceil(ratings.scoreSum / ratings.numRatings);
117 		return Array(avg).fill(5);
118 	}
