 94 		const url = `${this.apiUrl}/users`;
 95

103 	/**
104 		* Returns user with given userId if it exists in users array.
105 		* If no user with given userId found, returns null instead.
106 		* @param {number} userId The express routing HTTP client request object.
107 		* @return {User} The User object with the matching userId, or null if none found.
108 	*/
109 	getUser(userId: number): User {
110 		if (this.users === undefined || this.users === null || this.users.length === 0) {
111 			return null;
112 		}
113 		return this.users.find(cur => cur.userId === userId);
114 	}

216 	/**
217 		* Finds best matches of current user, then creates corresponding events and populates server.
218 		* Calls the following server-side APIs: GET /match/:userId, POST /event/:userId.
219 		* @param {number} userId The userId whose matches we are computing.
220 		* @return {Observable<Event[]>} An Observable object that promises population of events collection with appropriate events.
221 	*/
222 	fetchMatches(userId: number): Observable<Match[]> {
223 		const url = `${this.matchUrl}/${userId}`;
224 
225 		return this.http.get<Match[]>(url).pipe(
226 			tap(res => {
227 				for (let match of res) {
228 					let event: Event = {
229 						eventId: 0,
230 						acceptedIds: [],
231 						invitedIds: [match.match_id, userId],
232 						activity: match.event,
233 						startTime: match.unix_time,
234 						endTime: new Date(match.unix_time.getSeconds() + 180),
235 						status: "matched",
236 						location: match.location
237 					};
238 					this.http.post<Event>(`${this.apiUrl}/event/${userId}`, event).subscribe();
239 				}
240 			})
241 		);
242 	}

288 	/**
289 		* Deletes event from database with event id removeEventId. Deletes event from matchedEvents, pendingEvents, and confirmedEvents arrays.
290 		* Calls the following server-side API: DELETE /event/:removeEventId.
291 		* @param {number} removeEventId The event id of the event we are deleting.
292 		* @return {void}
293 	*/
294 	deleteEvent(removeEventId: number): void {
295 		const url = `${this.apiUrl}/event/${removeEventId}`;
296 
297 		let found: boolean = false;
298 		for(let i = 0; i < this.matchedEvents.length; i++) {
299 			if (this.matchedEvents[i].eventId == removeEventId) {
300 				found = true;
301 				this.matchedEvents.splice(i, 1);
302 			}
303 		}
304 
305 		if (!found) {
306 			for(let i = 0; i < this.pendingEvents.length; i++) {
307 				if (this.pendingEvents[i].eventId == removeEventId) {
308 					found = true;
309 					this.pendingEvents.splice(i, 1);
310 				}
311 			}
312 		}
313 
314 		if (!found) {
315 			for(let i = 0; i < this.confirmedEvents.length; i++) {
316 				if (this.confirmedEvents[i].eventId == removeEventId) {
317 					found = true;
318 					this.confirmedEvents.splice(i, 1);
319 				}
320 			}
321 		}
322 
323 		this.http.delete(url, options).subscribe();
324 		console.log('deleting event');
325 	}

327 	/**
328 		* Updates event in database whose event id matches updatedEvent.eventId. Updates matchedEvents, pendingEvents, and confirmedEvents arrays.
329 		* Calls the following server-side API: PUT /event/:updatedEvent.eventId, body: {event: updatedEvent}
330 		* @param {Event} updatedEvent The updated event that we are sending to the server.
331 		* @return {void}
332 	*/
333 	updateEvent(updatedEvent: Event): void {
334 		const url = `${this.apiUrl}/event/${updatedEvent.eventId}`;
335 		var update = {event: updatedEvent};
336 
337 		let found: boolean = false;
338 		for(let i = 0; i < this.matchedEvents.length; i++) {
339 			if (this.matchedEvents[i].eventId == updatedEvent.eventId) {
340 				found = true;
341 				this.pendingEvents.push(updatedEvent);
342 				this.matchedEvents.splice(i, 1);
343 			}
344 		}
345 
346 		if (!found) {
347 			for(let i = 0; i < this.pendingEvents.length; i++) {
348 				if (this.pendingEvents[i].eventId == updatedEvent.eventId) {
349 					found = true;
350 					this.confirmedEvents.push(updatedEvent);
351 					this.pendingEvents.splice(i, 1);
352 				}
353 			}
354 		}
355 
356 		this.http.put(url, update).subscribe();
357 		console.log('updating event');
358 	}
