  2 /**
  3 	* Function to return a JSON object containing the express router properties.
  4 	* @param {Object} req The express routing HTTP client request object.
  5 	* @param {Object} res The express routing HTTP client response object.
  6 	* @param {callback} next The express routing callback function to invoke next middleware in the stack.
  7 	* @return {Object} A JSON object that holds req, res, and next.
  8 */
  9 var routerProperties = function(req, res, next) {
 10 	return {
 11 		req: req,
 12 		res: res,
 13 		next: next
 14 	};
 15 }

 17 /**
 18 	* Function to search for all users meeting a criteria.
 19 	* The server returns a 404 error if no users matching criteria are found.
 20 	* @param {Object} properties JSON object containing the express router properties.
 21 	* @param {Object} criteria JSON object to determine the search criteria for the user(s) we wish to find.
 22 	* @param {callback} processUsers A callback function to be invoked on the resulting set of users found.
 23 	* @return {Void}
 24 */
 25 var searchUsers = function(properties, criteria, processUsers) {
 26 	let req = properties.req;
 27 	let res = properties.res;
 28 	let next = properties.next;
 29 
 30 	let db = req.app.locals.db;
 31 	const userCollection = db.collection("Users");
 32 
 33 	userCollection.find(criteria).toArray(function(err, users) {
 34 		if (err) {
 35 			next(err);
 36 			return;
 37 		}
 38 		if (users.length === 0) {
 39 			res.status(404).send(`Unable to find users that met search criteria`);
 40 			return;
 41 		}
 42 		processUsers(users);
 43 	});
 44 }

 83 /**
 84 	* Function to insert a new user into the Users collection in MongoDB.
 85 	* The server returns a 404 error if the email in the request body is already in use.
 86 	* Otherwise it creates a new user with the given userID and hashed password,
 87 	* default availabilities, and no activities or events, and adds it to the MongoDB database,
 88 	* returning a 201 status code.
 89 	* @param {Object} properties JSON object containing the express router properties.
 90 	* @param {string} email The email of the new user.
 91 	* @param {string} password The password of the new user that is encrypted with bcrypt.
 92 	* @return {Void}
 93 */
 94 var insertUser = function(properties, email, password) {
 95 	let req = properties.req;
 96 	let res = properties.res;
 97 	let next = properties.next;
 98 
 99 	let db = req.app.locals.db;
100 	const userCollection = db.collection("Users");
101 	const valuesCollection = db.collection("Values");
102 
103 	userCollection.find({"email": email}).toArray(
104 		function(err, result) {
105 			if (err) {
106 				next(err);
107 				return;
108 			}
109 			if (result.length !== 0) {
110 				res.status(404).send(`Email already in use`);
111 				return;
112 			}
113 
114 			valuesCollection.find({"name": "Users"}).toArray(function(err, resId) {
115 				if (err) {
116 					next(err);
117 					return;
118 				}
119 				let maxUserId = resId[0].maxUserId;
120 
121 				let defaultAvailability = [];
122 				for (let i = 0; i < 7; i++) {
123 					let week = [];
124 					for (let j = 0; j < 48; j++) {
125 						week.push(false);
126 					}
127 					defaultAvailability.push(week);
128 				}
129 
130 				let newUser = {
131 					userId: maxUserId,
132 					email: email,
133 					password: password,
134 					availability: defaultAvailability,
135 					activities: [],
136 					events: []
137 				};
138 				userCollection.insertOne(newUser, function (err, insertResult) {
139 					if (err) {
140 						next(err);
141 						return;
142 					}
143 					let newValue = {$set: {"maxUserId": maxUserId + 1}}; // this is buggy
144 					valuesCollection.updateOne({"name": "Users"}, newValue, function(err, updateResult) {
145 						if (err) {
146 							next(err);
147 							return;
148 						}
149 						res.status(201).render('profile', {
150 							userId: newUser.userId,
151 							email: newUser.email,
152 							activities: newUser.activities,
153 							availability: newUser.availability,
154 							events: newUser.events
155 						});
156 						return;
157 					});
158 				});
159 			});

161 		}
162 	);
