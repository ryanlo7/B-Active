 47 /**
 48 	* Compute a match score for two users.
 49 	* @param {Object} curr_user An object containing the user profile information for the logged
 50 	* in user.
 51 	* @param {Object} potential_match An object containing the user profile information for the the
 52 	* match candidate for the logged in user.
 53 	* @return {Array} An array of a matched activity and the match score.
 54 	* A score representing how good the match is between the logged in user and the potential match. 
 55 	* This match is based on availability, interest level, and skill level.
 56 */
 57 function matchUser(curr_user, potential_match) {
 58 	var availability_match_score = getAvailabilityMatchScore(curr_user["availability"], potential_match["availability"]);
 59 	var activity_match = getBestActivityMatch(curr_user["activities"], potential_match["activities"]);
 60 	var total_score = 0;
 61 
 62 	if (activity_match["interest_score"] != 0 && availability_match_score != 0) {
 63 		total_score = activity_match["interest_score"] + activity_match["skill_score"] + availability_match_score;
 64 	}
 65 	return [activity_match["name"], total_score];
 66 }

 69 /**
 70 	* Returns a score for the availability match between two users. 
 71 	* This is done by computing the largest overlapping block of time between the two users.
 72 	* This score is capped at a maximum defined by a constant MAX_AVAILABILITY_SCORE. 
 73 	* @param {Array<Array<<boolean>>} curr_user_availability A list of true/false availabilities
 74 	* for the current user for each thirty-minute time slot on each day of the week.
 75 	* @param {Array<Array<<boolean>>} potential_match_availability A list of true/false availabilities
 76 	* for the potentialMatch for each thirty-minute time slot on each day of the week.
 77 	* @return {number} A score representing how good the time availability match is, where 0 represents
 78 	* a failed match and MAX_AVAILABILITY_SCORE is the highest possible match score for this category.
 79 */
 80 
 81 function getAvailabilityMatchScore(curr_user_availability, potential_match_availability) {
 82 	var num_overlapping_periods = getAvailabilityMatch(curr_user_availability, potential_match_availability);
 83 
 84 	// Thirty minutes is two short for an activity, so a match requires at least an hour of matched times.
 85 	// Furthermore, activity matches are not expected to have an activity for over three hours, 
 86 	// so a maximum cutoff is set for highest possible match.
 87 	if (num_overlapping_periods <= 1) {
 88 		num_overlapping_periods = 0;
 89 	} else if (num_overlapping_periods > MAX_AVAILABILITY_SCORE) {
 90 		num_overlapping_periods = MAX_AVAILABILITY_SCORE;
 91 	}
 92 	return computeNormalizedScore(num_overlapping_periods, MAX_AVAILABILITY_SCORE);
 93 }

 96 /**
 97 	* Returns maximum number of consecutive overlapping half-hours for two users. This could
 98 	* span over multiple days of the week, as long as it is a consecutive chunk of hours (i.e.
 99 	* 11:00 PM on Monday until 1 AM on Tuesday).
100 	* @param {Array<Array<<boolean>>} curr_user_availability A list of true/false availabilities
101 	* for the current user for each thirty-minute time slot on each day of the week.
102 	* @param {Array<Array<<boolean>>} potential_match_availability A list of true/false availabilities
103 	* for the potentialMatch for each thirty-minute time slot on each day of the week.
104 	* @return {number} The maximum number of overlapping consecutive half-hours between the two users.
105 */
106 function getAvailabilityMatch(curr_user_availability, potential_match_availability) {
107 	var max_sequence = 0;
108 	var curr_sequence = 0;
109 
110 	for (var i = 0; i < curr_user_availability.length; i++) {
111 		for (var j = 0; j < curr_user_availability[i].length; j++) {
112 			if (curr_user_availability[i][j] && potential_match_availability[i][j]) {
113 				curr_sequence++;
114 				if (curr_sequence > max_sequence) {
115 					max_sequence = curr_sequence;
116 				}
117 			}
118 		 	else {
119 				curr_sequence = 0;
120 			}
121 		}
122 	}
123 
124 	return max_sequence;
125 }

127 /**
128 	* Returns an object containing the best activity match between two users and the
129 	* interest and skill level scores for that activity.
130 	* @param {Array<Object>} curr_user_activities A list of activities for the current user with
131 	* interest and skill level scores.
132 	* @param {Array<Object>} potential_match_activities A list of activities for the potential
133 	* match user with interest and skill level scores.
134 	* @return {Object} The matched activity yielding the highest score for the current user.
135 	* The object contains a field for the activity name, a field for the interest match score, and
136 	* a field for the skill match score.
137 */
138 function getBestActivityMatch(curr_user_activities, potential_match_activities) {
139 	return getBestActivityMatchFromList(generateActivityMatches(curr_user_activities, potential_match_activities));
140 }

142 /**
143 	* Generates a list of activity matches with an interest match and a skill match for each activity.
144 	* @param {Array<Object>} curr_user_activities A list of activities for the current user with
145 	* interest and skill level scores.
146 	* @param {Array<Object>} potential_match_activities A list of activities for the potential
147 	* match user with interest and skill level scores.
148 	* @return {Array<Object>} List of objects with the potential match activities. Each object has
149 	* the name of the potential match activity, a skill score, and an interest score.
150 */
151 function generateActivityMatches(curr_user_activities, potential_match_activities) {
152 	var activity_matches = []
153 
154 	// Create a list with all activities with match score temporarily as 0.
155 	for (var i = 0; i < curr_user_activities.length; i++) {
156 		var activity_match = {};
157 		activity_match["name"] = curr_user_activities[i]["name"];
158 		activity_match["skill_score"] = 0;
159 		activity_match["interest_score"] = 0;
160 		activity_matches.push(activity_match);
161 	}
162 
163 	// Fill array of matched activities.
164 	for (var i = 0; i < curr_user_activities.length; i++) {
165 		for (var j = 0; j < potential_match_activities.length; j++) {
166 			if (curr_user_activities[i]["name"] !== potential_match_activities[i]["name"]) {
167 				continue;
168 			}
169 			activity_matches[i]["skill_score"] = computeSkillMatch(curr_user_activities[i]["skill"], potential_match_activities[i]["skill"]);
170 			activity_matches[i]["interest_score"] = computeInterestMatch(curr_user_activities[i]["interest"], potential_match_activities[i]["interest"]);
171 		}
172 	}
173 	return activity_matches;
174 }

176 /**
177 	* Returns an object containing the best activity match between two users and the
178 	* interest and skill level scores for that activity, given a list of activities and scores.
179 	* The best match is defined as having the highest skill score and interest score combined, 
180 	* disregarding activities with an interest match of 0.
181 	* @param {Array<Object>} activity_matches Current activity matches with name, skill score, and interest
182 	* score for each activity.
183 	* @return {Object} The matched activity yielding the highest score for the current user.
184 	* The object contains a field for the activity name, a field for the interest match score, and
185 	* a field for the skill match score.
186 */
187 function getBestActivityMatchFromList(activity_matches) {
188 	var best_activity_match = {};
189 	// Default initialization so that keys are present in the object. When using the score in other functions,
190 	// handle exceptional case of having 0 interest match score (meaning no match).
191 	best_activity_match["name"] = "lifting";
192 	best_activity_match["skill_score"] = 0;
193 	best_activity_match["interest_score"] = 0;
194 	var max_score = 0;
195 
196 	for (var i = 0; i < activity_matches.length; i++) {
197 		if (activity_matches[i]["interest_score"] == 0) {
198 			continue;
199 		} else {
200 			var curr_score = activity_matches[i]["interest_score"] + activity_matches[i]["skill_score"];
201 			if (curr_score > max_score) {
202 				max_score = curr_score;
203 				best_activity_match["name"] = activity_matches[i]["name"];
204 				best_activity_match["skill_score"] = activity_matches[i]["skill_score"];
205 				best_activity_match["interest_score"] = activity_matches[i]["interest_score"];
206 			}
207 		}
208 	}
209 	return best_activity_match;
210 }

212 /**
213 	* Returns a score for interest match. The computation most highly weights the current
214 	* user's interests, and then weights the other user's interest level with secondary importance.
215 	* Note that an interest level of 1 is the lowest possible interest for an activity, so this
216 	* automatically results in a score of 0 for the interest match.
217 	* @param {number} curr_user_interest Interest level for current user
218 	* @param {number} potential_match_interest Interest level for other user
219 	* @return {number} A normalized score representing the interest match between two users.
220 */
221 function computeInterestMatch(curr_user_interest, potential_match_interest) {
222 	var unnormalized_score = curr_user_interest + 0.5*(potential_match_interest);
223 	if (curr_user_interest == 1 || potential_match_interest == 1) {
224 		unnormalized_score = 0;
225 	}
226 	return computeNormalizedScore(unnormalized_score, MAX_INTEREST_SCORE);
227 }
228 
229 /**
230 	* Returns a score for skill match. The score is high if the two users have similar skill levels.
231 	* @param {number} curr_user_skill Skill level of current user
232 	* @param {number} potential_match_skill Skill level of potential match
233 	* @return {number} A normalized score representing the skill level match between two users.
234 */
235 function computeSkillMatch(curr_user_skill, potential_match_skill) {
236 	var unnormalized_score = MAX_SKILL_SCORE - Math.abs(curr_user_skill-potential_match_skill)
237 	return computeNormalizedScore(unnormalized_score, MAX_SKILL_SCORE);
238 }
239 
240 /**
241 	* Normalizes the score for the three categories, so all three types of scores
242 	* have the same maximum and minimum values.
243 	* @param {number} curr_score Current score for that category
244 	* @param {number} curr_max Maximum score for that cateogry
245 	* @return {number} A normalized score representing the match score between two users for one category.
246 */
247 function computeNormalizedScore(curr_score, curr_max) {
248 	return curr_score/curr_max*NORMALIZED_BASE;
249 }
