1 var express = require('express');
  2 var router = express.Router();
  3 
  4 /* GET users listing. */
  5 router.get('/:userid',
  6 	/**
  7 		* Function to handle GET requests for user events page.
  8 		* If the userID is found, it renders the events page for the user with that ID.
  9 		* @param {Object} req The express routing HTTP client request object,
 10 		* which contains the userID as a parameter.
 11 		* @param {Object} res The express routing HTTP client response object.
 12 		* @param {callback} next The express routing callback function to invoke next middleware in the stack.
 13 		* @return {Object} A JSON object that holds req, res, and next.
 14 	*/
 15 	function(req, res, next) {
 16 	var db = req.app.locals.db; //get instance of db
 17 	var userId = parseInt(req.params.userid);
 18 
 19 	db.collection('Users')
 20 			.find({'userId': userId})
 21 			.toArray(function(err, results) {
 22 				if (results.length == 0) {
 23 					res.status(404).send("404: userId not found");
 24 				} else {
 25 					user = results[0]; // should only be one match
 26 
 27 					res.render('events', {
 28 					 	userId: userId,
 29 						events: user.events
 30 					});
 31 				}
 32 			});
 33 
 34 });
 35 
 36 module.exports = router;
