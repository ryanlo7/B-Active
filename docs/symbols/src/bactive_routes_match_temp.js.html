 19 /**
 20 	* Compute a match score for two users.
 21 	* @param {Object} curr_user An object containing the user profile information for the logged
 22 	* in user.
 23 	* @param {Object} potential_match An object containing the user profile information for the the
 24 	* match candidate for the logged in user.
 25 	* @return {Array} An array of a matched activity and the match score.
 26 	* A score representing how good the match is between the logged in user and the potential match. 
 27 	* This match is based on availability, interest level, and skill level.
 28 */
 29 function matchUser(curr_user, potential_match) {
 30 	var availability_match_score = getAvailabilityMatchScore(curr_user["availability"], potential_match["availability"]);
 31 	var activity_match = getBestActivityMatch(curr_user["activities"], potential_match["activities"]);
 32 	var total_score = 0;
 33 
 34 	if (activity_match["interest_score"] != 0 && availability_match_score != 0) {
 35 		total_score = activity_match["interest_score"] + activity_match["skill_score"] + availability_match_score;
 36 	}
 37 	return [activity_match["name"], total_score];
 38 }

 41 /**
 42 	* Returns a score for the availability match between two users. 
 43 	* This is done by computing the largest overlapping block of time between the two users.
 44 	* This score is capped at a maximum defined by a constant MAX_AVAILABILITY_SCORE. 
 45 	* @param {Array<Array<<boolean>>} curr_user_availability A list of true/false availabilities
 46 	* for the current user for each thirty-minute time slot on each day of the week.
 47 	* @param {Array<Array<<boolean>>} potential_match_availability A list of true/false availabilities
 48 	* for the potentialMatch for each thirty-minute time slot on each day of the week.
 49 	* @return {number} A score representing how good the time availability match is, where 0 represents
 50 	* a failed match and MAX_AVAILABILITY_SCORE is the highest possible match score for this category.
 51 */
 52 
 53 function getAvailabilityMatchScore(curr_user_availability, potential_match_availability) {
 54 	var num_overlapping_periods = getAvailabilityMatch(curr_user_availability, potential_match_availability);
 55 
 56 	// Thirty minutes is two short for an activity, so a match requires at least an hour of matched times.
 57 	// Furthermore, activity matches are not expected to have an activity for over three hours, 
 58 	// so a maximum cutoff is set for highest possible match.
 59 	if (num_overlapping_periods <= 1) {
 60 		num_overlapping_periods = 0;
 61 	} else if (num_overlapping_periods > MAX_AVAILABILITY_SCORE) {
 62 		num_overlapping_periods = MAX_AVAILABILITY_SCORE;
 63 	}
 64 	return computeNormalizedScore(num_overlapping_periods, MAX_AVAILABILITY_SCORE);
 65 }

 68 /**
 69 	* Returns maximum number of consecutive overlapping half-hours for two users. This could
 70 	* span over multiple days of the week, as long as it is a consecutive chunk of hours (i.e.
 71 	* 11:00 PM on Monday until 1 AM on Tuesday).
 72 	* @param {Array<Array<<boolean>>} curr_user_availability A list of true/false availabilities
 73 	* for the current user for each thirty-minute time slot on each day of the week.
 74 	* @param {Array<Array<<boolean>>} potential_match_availability A list of true/false availabilities
 75 	* for the potentialMatch for each thirty-minute time slot on each day of the week.
 76 	* @return {number} The maximum number of overlapping consecutive half-hours between the two users.
 77 */
 78 function getAvailabilityMatch(curr_user_availability, potential_match_availability) {
 79 	var max_sequence = 0;
 80 	var curr_sequence = 0;
 81 
 82 	for (var i = 0; i < curr_user_availability.length; i++) {
 83 		for (var j = 0; j < curr_user_availability[i].length; j++) {
 84 			if (curr_user_availability[i][j] && potential_match_availability[i][j]) {
 85 				curr_sequence++;
 86 				if (curr_sequence > max_sequence) {
 87 					max_sequence = curr_sequence;
 88 				}
 89 			}
 90 		 	else {
 91 				curr_sequence = 0;
 92 			}
 93 		}
 94 	}
 95 
 96 	return max_sequence;
 97 }

 99 /**
100 	* Returns an object containing the best activity match between two users and the
101 	* interest and skill level scores for that activity.
102 	* @param {Array<Object>} curr_user_activities A list of activities for the current user with
103 	* interest and skill level scores.
104 	* @param {Array<Object>} potential_match_activities A list of activities for the potential
105 	* match user with interest and skill level scores.
106 	* @return {Object} The matched activity yielding the highest score for the current user.
107 	* The object contains a field for the activity name, a field for the interest match score, and
108 	* a field for the skill match score.
109 */
110 function getBestActivityMatch(curr_user_activities, potential_match_activities) {
111 	return getBestActivityMatchFromList(generateActivityMatches(curr_user_activities, potential_match_activities));
112 }

114 /**
115 	* Generates a list of activity matches with an interest match and a skill match for each activity.
116 	* @param {Array<Object>} curr_user_activities A list of activities for the current user with
117 	* interest and skill level scores.
118 	* @param {Array<Object>} potential_match_activities A list of activities for the potential
119 	* match user with interest and skill level scores.
120 	* @return {Array<Object>} List of objects with the potential match activities. Each object has
121 	* the name of the potential match activity, a skill score, and an interest score.
122 */
123 function generateActivityMatches(curr_user_activities, potential_match_activities) {
124 	var activity_matches = []
125 
126 	// Create a list with all activities with match score temporarily as 0.
127 	for (var i = 0; i < curr_user_activities.length; i++) {
128 		var activity_match = {};
129 		activity_match["name"] = curr_user_activities[i]["name"];
130 		activity_match["skill_score"] = 0;
131 		activity_match["interest_score"] = 0;
132 		activity_matches.push(activity_match);
133 	}
134 
135 	// Fill array of matched activities.
136 	for (var i = 0; i < curr_user_activities.length; i++) {
137 		for (var j = 0; j < potential_match_activities.length; j++) {
138 			if (curr_user_activities[i]["name"] !== potential_match_activities[i]["name"]) {
139 				continue;
140 			}
141 			activity_matches[i]["skill_score"] = computeSkillMatch(curr_user_activities[i]["skill"], potential_match_activities[i]["skill"]);
142 			activity_matches[i]["interest_score"] = computeInterestMatch(curr_user_activities[i]["interest"], potential_match_activities[i]["interest"]);
143 		}
144 	}
145 	return activity_matches;
146 }

148 /**
149 	* Returns an object containing the best activity match between two users and the
150 	* interest and skill level scores for that activity, given a list of activities and scores.
151 	* The best match is defined as having the highest skill score and interest score combined, 
152 	* disregarding activities with an interest match of 0.
153 	* @param {Array<Object>} activity_matches Current activity matches with name, skill score, and interest
154 	* score for each activity.
155 	* @return {Object} The matched activity yielding the highest score for the current user.
156 	* The object contains a field for the activity name, a field for the interest match score, and
157 	* a field for the skill match score.
158 */
159 function getBestActivityMatchFromList(activity_matches) {
160 	var best_activity_match = {};
161 	// Default initialization so that keys are present in the object. When using the score in other functions,
162 	// handle exceptional case of having 0 interest match score (meaning no match).
163 	best_activity_match["name"] = "lifting";
164 	best_activity_match["skill_score"] = 0;
165 	best_activity_match["interest_score"] = 0;
166 	var max_score = 0;
167 
168 	for (var i = 0; i < activity_matches.length; i++) {
169 		if (activity_matches[i]["interest_score"] == 0) {
170 			continue;
171 		} else {
172 			var curr_score = activity_matches[i]["interest_score"] + activity_matches[i]["skill_score"];
173 			if (curr_score > max_score) {
174 				max_score = curr_score;
175 				best_activity_match["name"] = activity_matches[i]["name"];
176 				best_activity_match["skill_score"] = activity_matches[i]["skill_score"];
177 				best_activity_match["interest_score"] = activity_matches[i]["interest_score"];
178 			}
179 		}
180 	}
181 	return best_activity_match;
182 }

184 /**
185 	* Returns a score for interest match. The computation most highly weights the current
186 	* user's interests, and then weights the other user's interest level with secondary importance.
187 	* Note that an interest level of 1 is the lowest possible interest for an activity, so this
188 	* automatically results in a score of 0 for the interest match.
189 	* @param {number} curr_user_interest Interest level for current user
190 	* @param {number} potential_match_interest Interest level for other user
191 	* @return {number} A normalized score representing the interest match between two users.
192 */
193 function computeInterestMatch(curr_user_interest, potential_match_interest) {
194 	var unnormalized_score = curr_user_interest + 0.5*(potential_match_interest);
195 	if (curr_user_interest == 1 || potential_match_interest == 1) {
196 		unnormalized_score = 0;
197 	}
198 	return computeNormalizedScore(unnormalized_score, MAX_INTEREST_SCORE);
199 }

201 /**
202 	* Returns a score for skill match. The score is high if the two users have similar skill levels.
203 	* @param {number} curr_user_skill Skill level of current user
204 	* @param {number} potential_match_skill Skill level of potential match
205 	* @return {number} A normalized score representing the skill level match between two users.
206 */
207 function computeSkillMatch(curr_user_skill, potential_match_skill) {
208 	var unnormalized_score = MAX_SKILL_SCORE - Math.abs(curr_user_skill-potential_match_skill)
209 	return computeNormalizedScore(unnormalized_score, MAX_SKILL_SCORE);
210 }

212 /**
213 	* Normalizes the score for the three categories, so all three types of scores
214 	* have the same maximum and minimum values.
215 	* @param {number} curr_score Current score for that category
216 	* @param {number} curr_max Maximum score for that cateogry
217 	* @return {number} A normalized score representing the match score between two users for one category.
218 */
219 function computeNormalizedScore(curr_score, curr_max) {
220 	return curr_score/curr_max*NORMALIZED_BASE;
221 }
